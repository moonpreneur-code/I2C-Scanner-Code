I2C LCD Library code


from machine import Pin, I2C

# Initialize I2C on GP1 (SCL) and GP0 (SDA)
i2c = I2C(0, scl=Pin(5), sda=Pin(4))

# Scan for I2C devices and print their addresses in hexadecimal
devices = i2c.scan()

if devices:
    print("I2C device(s) found:")
    for device in devices:
        print("Address in HEX: {}".format(hex(device)))
else:
    print("No I2C devices found")
